stages:
  - vps

free-vps:
  stage: vps
  image: ubuntu:22.04
  timeout: 5h 50m

  script:
    - apt-get update -qq
    - apt-get install -y openssh-server curl sudo

    - useradd -m -s /bin/bash vpsuser || true
    - echo "vpsuser:Vps@2024#Secure" | chpasswd
    - echo "vpsuser ALL=(ALL) NOPASSWD:ALL" >> /etc/sudoers

    - mkdir -p /run/sshd
    - echo "PasswordAuthentication yes" >> /etc/ssh/sshd_config
    - echo "PermitRootLogin yes" >> /etc/ssh/sshd_config
    - /usr/sbin/sshd

    - curl -L https://github.com/cloudflare/cloudflared/releases/latest/download/cloudflared-linux-amd64 -o cloudflared
    - chmod +x cloudflared
    - mv cloudflared /usr/local/bin/

    - cloudflared tunnel --url ssh://localhost:22 > /tmp/cf.log 2>&1 &
    - sleep 10

    - |
      CF_URL=$(grep -oP 'https://[a-zA-Z0-9\-]+\.trycloudflare\.com' /tmp/cf.log | head -1)
      echo "╔════════════════════════════════════════════╗"
      echo "║           🚀 VPS LOGIN DETAILS             ║"
      echo "╠════════════════════════════════════════════╣"
      echo "║ 🌐 URL      : $CF_URL"
      echo "║ 👤 Username : vpsuser"
      echo "║ 🔑 Password : Vps@2024#Secure"
      echo "╠════════════════════════════════════════════╣"
      echo "║ 💻 SSH CMD:"
      echo "║ ssh -o ProxyCommand='cloudflared access ssh --hostname $CF_URL' vpsuser@$CF_URL"
      echo "╚════════════════════════════════════════════╝"

    - sleep 20000

  only:
    - web
